REACT & REACT STYLED COMPONENTS:

This Snippet includes Reactjs, React Context, React Styled Components for authentication request of a login page


import React, { useState, useContext } from "react";

import { LoginContext } from "../contexts/LoginContext";

import styled from "styled-components";

const FormWrapper = styled.form`

 max-width: 400px;

 margin: auto;

 padding: 20px;

 border: 1px solid #ccc;

 border-radius: 8px;

`;

const InputGroup = styled.div`

 margin-bottom: 15px;

`;

const Label = styled.label`

 display: block;

 margin-bottom: 5px;

`;

const Input = styled.input`

 width: 100%;

 padding: 10px;

 border: 1px solid #ccc;

 border-radius: 4px;

`;

const Button = styled.button`

 width: 100%;

 padding: 10px;

 background-color: #007bff;

 color: white;

 border: none;

 border-radius: 4px;

 cursor: pointer;

 &:disabled {

   background-color: #ccc;

   cursor: not-allowed;

 }

`;

function Login() {

 const [formData, setFormData] = useState({ email: "", password: "" });

 const { login, loading, error, user } = useContext(LoginContext);

 const handleChange = (e) => {

   setFormData({ ...formData, [e.target.name]: e.target.value });

 };

 const handleSubmit = (e) => {

   e.preventDefault();

   login(formData);

 };

 return (

   <FormWrapper onSubmit={handleSubmit}>

     {error && <p style={{ color: "red" }}>{error}</p>}

     {user && <p style={{ color: "green" }}>Login successful!</p>}

     <InputGroup>

       <Label>Email</Label>

       <Input

         type="email"

         name="email"

         placeholder="Enter your email"

         value={formData.email}

         onChange={handleChange}

         required

       />

     </InputGroup>

     <InputGroup>

       <Label>Password</Label>

       <Input

         type="password"

         name="password"

         placeholder="Enter your password"

         value={formData.password}

         onChange={handleChange}

         required

       />

     </InputGroup>

     <Button type="submit" disabled={loading}>

       {loading ? "Logging in..." : "Login"}

     </Button>

   </FormWrapper>

 );

}

export default Login;

REACT CONTEXT & REACT QUERY:

Here is React Context setup using react query:

import React, { createContext, useContext } from "react";

import { useMutation, QueryClient, QueryClientProvider } from "react-query";

const LoginContext = createContext();

export const LoginProvider = ({ children }) => {

 const queryClient = new QueryClient();

 const loginMutation = useMutation(

   async (formData) => {

     const response = await fetch(`${process.env.REACT_APP_LARAVEL_LOGIN_API}/login`, {

       method: "POST",

       headers: { "Content-Type": "application/json" },

       body: JSON.stringify(formData),

     });

     if (!response.ok) {

       throw new Error("Invalid login credentials");

     }

     return response.json();

   }

 );

 const login = async (formData) => {

   try {

     await loginMutation.mutateAsync(formData);

   } catch (err) {

     // Handle error in component

   }

 };

 return (

   <QueryClientProvider client={queryClient}>

     <LoginContext.Provider value={{ login, ...loginMutation }}>

       {children}

     </LoginContext.Provider>

   </QueryClientProvider>

 );

};

export const useLogin = () => {

 return useContext(LoginContext);

};

LARAVEL:

Laravel Controller Login method to handle login logic from backend:

use Illuminate\Support\Facades\Auth;

public function login(Request $request) {

   $credentials = $request->only('email', 'password');

   if (Auth::attempt($credentials)) {

       $user = Auth::user();

       return response()->json(['user' => $user], 200);

   }

   return response()->json(['error' => 'Invalid login credentials'], 401);

}

DOCKER: (using redis)

Using Docker to create Redis cache for scalability in microservices:

import redis from 'redis';

import dotenv from 'dotenv';

// Load environment variables from .env file

dotenv.config();

let redisClient;

export const connectRedis = async () => {

 if (redisClient) return redisClient;

 try {

   const redisUrl = `${process.env.REDIS_HOST}:${process.env.REDIS_PORT}`;

   console.log('Connecting to Redis with URL:', redisUrl);

   redisClient = redis.createClient({

     url: redisUrl,

     socket: {

       tls: process.env.REDIS_TLS === 'true',

     },

   });

   redisClient.on('error', (error) => {

     console.error('Error connecting to Redis:', error);

   });

   await redisClient.connect();

   redisClient.on('connect', () => {

     console.log('Connected to Redis');

   });

   redisClient.on('ready', () => {

     console.log('Redis is ready to be used');

   });

   redisClient.on('end', () => {

     console.log('Redis connection has ended');

   });

 } catch (error) {

   console.error('Failed to connect to Redis:', error);

 }

 return redisClient;

};

export const getRedisClient = () => redisClient;
